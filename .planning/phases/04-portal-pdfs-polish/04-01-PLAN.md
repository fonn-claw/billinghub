```typescript
import { db } from "@/lib/db";
import { customers, invoices, payments } from "@/lib/db/schema";
import { eq, sql, and, ne, asc } from "drizzle-orm";

export async function getCustomerByUserId(userId: string) {
  const [customer] = await db
    .select()
    .from(customers)
    .where(eq(customers.userId, userId))
    .limit(1);
  return customer ?? null;
}

export async function getPortalSummary(customerId: string) {
  // Reuse getCustomerBalance pattern but also get next due date and last payment date
  // ... currentBalance, overdueAmount from getCustomerBalance pattern
  // ... nextDueDate: earliest unpaid invoice due date
  // ... lastPaymentDate: most recent payment date
}
```

The `getPortalSummary` function should return:
- `currentBalance: number` (totalCharged - totalPaid, in cents)
- `overdueAmount: number` (overdue invoices balance, in cents)
- `nextDueDate: string | null` (earliest due date among unpaid non-draft invoices, ISO string)
- `lastPaymentDate: string | null` (most recent payment date, ISO string)
- `invoiceCount: number` (total non-draft invoices)

Query pattern for nextDueDate: SELECT MIN(due_date) from invoices WHERE customer_id = X AND status NOT IN ('draft', 'paid') AND due_date >= current_date.
Query pattern for lastPaymentDate: SELECT MAX(payment_date) from payments JOIN invoices ON payments.invoice_id = invoices.id WHERE invoices.customer_id = X.

```typescript
export function InvoicePDFLink({ invoice }: { invoice: InvoiceDetail }) {

  );
}
```
